ผู้ดูแลระบบ
function startExam() { const name = document.getElementById("studentName").value.trim(); const school = document.getElementById("schoolName").value; if (name === "" || school === "") { alert("กรุณากรอกชื่อและเลือกโรงเรียน"); return; } const currentUser = { userId: "SPS-" + Date.now(), fullname: name, school: school, examName: "โรงเรียนส่งเสริมความปลอดภัย (SPS)", examTime: 30, passScore: 16, startTime: new Date().toLocaleString("th-TH") }; localStorage.setItem( "currentUser", JSON.stringify(currentUser) ); window.location.href = "exam-room.html"; }
